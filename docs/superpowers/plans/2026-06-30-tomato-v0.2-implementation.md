# Tomato v0.2.0 Usability Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Make tomato reliable for daily use by adding workflow resume/rerun controls, custom user steps, artifact diffing, and release channel polish.

**Architecture:** Keep tomato pure CLI. Add small focused packages for run state (`pkg/state`) and custom steps (`pkg/customstep`), then wire them into the existing `pkg/engine`, `cmd`, and `pkg/history` flows. Preserve the file-in/file-out model so future remote agents remain viable.

**Tech Stack:** Go 1.26, Cobra CLI, yaml.v3, standard library (`os`, `filepath`, `encoding/json`, `os/exec`).

## File Structure

### New files

- `pkg/state/state.go` — persists workflow progress to `.tomato/state/<workflow>.json`; supports failed step, completed steps, resume metadata.
- `pkg/state/state_test.go` — tests read/write/clear state.
- `pkg/customstep/customstep.go` — executes user-defined `custom_steps` from `tomato.yaml` using prompt/input/output/model declarations.
- `pkg/customstep/customstep_test.go` — tests prompt rendering, file glob resolution, output writing.
- `pkg/history/diff.go` — compares artifacts between two run snapshots or filesystem paths.
- `pkg/history/diff_test.go` — tests text diff output.

### Modified files

- `pkg/config/config.go` — add `CustomSteps map[string]CustomStepDef`; add parse validation for custom step names and v0.2 workflow steps.
- `pkg/engine/engine.go` — add `RunOptions{From, Resume, Force}`; integrate state persistence, `--from`, `--resume`, and custom step dispatch.
- `cmd/commands.go` — add `tomato run --from`, `tomato run --resume`; add `tomato history diff <run-a> <run-b> <artifact>`.
- `cmd/helpers.go` — propagate run options into engine; reuse existing `--feature` and `--force` handling.
- `pkg/runner/runner.go` — snapshot output artifacts into `.tomato/runs/<id>/artifacts/` so `history diff` has stable inputs.
- `.github/workflows/release.yml` — split stable tag releases from main build releases.
- `install.sh` — add `CHANNEL=stable|nightly` and choose release tag accordingly.
- `README.md` — document v0.2 flags, custom steps, history diff, and install channels.

```go
package state

import (
	"encoding/json"
	"fmt"
	"os"
	"path/filepath"
	"strings"
	"time"
)

// WorkflowState records resumable workflow progress.
type WorkflowState struct {
	Workflow       string    `json:"workflow"`
	Feature        string    `json:"feature"`
	CurrentStep    string    `json:"current_step"`
	FailedStep     string    `json:"failed_step,omitempty"`
	CompletedSteps []string  `json:"completed_steps"`
	LastRunID      string    `json:"last_run_id,omitempty"`
	UpdatedAt      time.Time `json:"updated_at"`
}

func Path(repoDir, workflow, feature string) string {
	name := sanitize(workflow) + "-" + sanitize(feature) + ".json"
	return filepath.Join(repoDir, ".tomato", "state", name)
}

func Save(repoDir string, s WorkflowState) error {
	s.UpdatedAt = time.Now().UTC()
	path := Path(repoDir, s.Workflow, s.Feature)
	if err := os.MkdirAll(filepath.Dir(path), 0755); err != nil {
		return err
	}
	data, err := json.MarshalIndent(s, "", "  ")
	if err != nil {
		return err
	}
	return os.WriteFile(path, data, 0644)
}

func Load(repoDir, workflow, feature string) (*WorkflowState, error) {
	path := Path(repoDir, workflow, feature)
	data, err := os.ReadFile(path)
	if err != nil {
		return nil, fmt.Errorf("loading workflow state %s: %w", path, err)
	}
	var s WorkflowState
	if err := json.Unmarshal(data, &s); err != nil {
		return nil, err
	}
	return &s, nil
}

func Clear(repoDir, workflow, feature string) error {
	path := Path(repoDir, workflow, feature)
	if err := os.Remove(path); err != nil && !os.IsNotExist(err) {
		return err
	}
	return nil
}

func sanitize(s string) string {
	s = strings.ReplaceAll(s, "/", "-")
	s = strings.ReplaceAll(s, "\\", "-")
	s = strings.ReplaceAll(s, " ", "-")
	if s == "" {
		return "default"
	}
	return s
}
```

```go
type RunOptions struct {
	From   string
	Resume bool
	Force  bool

func (e *Engine) planSteps(workflowName string, opts RunOptions) []config.WorkflowStep {
	steps, _ := e.planStepsChecked(workflowName, opts)
	return steps

func (e *Engine) planStepsChecked(workflowName string, opts RunOptions) ([]config.WorkflowStep, error) {
	wf, ok := e.Workflows[workflowName]
	if !ok {
		return nil, fmt.Errorf("workflow %q not found", workflowName)
	}
	if opts.From == "" {
		return wf.Steps, nil
	}
	for i, s := range wf.Steps {
		if s.Name == opts.From {
			return wf.Steps[i:], nil
		}
	}
	return nil, fmt.Errorf("--from step %q not found in workflow %q", opts.From, workflowName)
}
```

```go
package customstep

import (
	"fmt"
	"os"
	"path/filepath"

	"github.com/bavocado/tomato/pkg/config"
	"github.com/bavocado/tomato/pkg/model"
	"github.com/bavocado/tomato/pkg/runner"
)

type Config struct {
	RepoDir string
	FeatureDir string
	ModelName string
	LLMStream runner.LLMFunc
	BudgetTracker *budget.Tracker
}

func Run(name string, def config.CustomStepDef, cfg Config) *model.StepResult {
	promptBytes, err := os.ReadFile(filepath.Join(cfg.RepoDir, def.Prompt))
	if err != nil {
		return &model.StepResult{StepName: name, Success: false, Error: fmt.Sprintf("reading prompt %s: %v", def.Prompt, err)}
	}
	inputs := expandInputs(cfg.RepoDir, def.Inputs)
	outputs := make([]string, 0, len(def.Outputs))
	for _, out := range def.Outputs { outputs = append(outputs, filepath.Join(cfg.RepoDir, out)) }
	return runner.Execute(name, string(promptBytes), inputs, outputs, cfg.RepoDir, cfg.ModelName, cfg.LLMStream, "custom-v1", cfg.BudgetTracker)
}

func expandInputs(repoDir string, patterns []string) []string {
	var out []string
	for _, p := range patterns {
		matches, _ := filepath.Glob(filepath.Join(repoDir, p))
		out = append(out, matches...)
	}
	return out
}
```

```go
package history

import (
	"fmt"
	"os"
	"path/filepath"
	"strings"
)

func Diff(repoDir, runA, runB, artifact string) (string, error) {
	a := filepath.Join(repoDir, ".tomato", "runs", runA, "artifacts", artifact)
	b := filepath.Join(repoDir, ".tomato", "runs", runB, "artifacts", artifact)
	ad, err := os.ReadFile(a); if err != nil { return "", err }
	bd, err := os.ReadFile(b); if err != nil { return "", err }
	return simpleDiff(string(ad), string(bd)), nil
}

func simpleDiff(a, b string) string {
	if a == b { return "(no changes)\n" }
	al := strings.Split(a, "\n")
	bl := strings.Split(b, "\n")
	var out strings.Builder
	fmt.Fprintf(&out, "--- old\n+++ new\n")
	max := len(al); if len(bl) > max { max = len(bl) }
	for i := 0; i < max; i++ {
		var av, bv string
		if i < len(al) { av = al[i] }
		if i < len(bl) { bv = bl[i] }
		if av != bv {
			if av != "" { fmt.Fprintf(&out, "- %s\n", av) }
			if bv != "" { fmt.Fprintf(&out, "+ %s\n", bv) }
		}
	}
	return out.String()
}
```
